sudo apt-get install freepats

sudo apt-get install Fluidr3

# Correr midi de allegro a traves de timidity:
## Cargar modulos
sudo modprobe snd-seq snd-virmidi
## Corremos timidity *en otra consola* (se puede cargar al inicio en segundo plano):
timidity -iA
## Buscamos puertos MIDI disponibles:
aconnect -i
## Conectamos un Virtual Raw MIDI port al timidity (reemplazar XX por el puerto correspondinte):
aconnect XX:0 128:0
## Buscamos nombre del dispositivo Virtual Raw MIDI port al que fue conectado:
amidi -l
## Creamos archivo (si no existe) /etc/allegro.cfg y agregamos:
[sound]
alsa_rawmidi_device = hw:1,0
